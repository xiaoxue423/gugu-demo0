# 快速上手

# 引入必要资源

import {Button} from "gugu-demo";
import "gugu-demo/dist/index.css";

# 注册组件

export default {
  name: "ComponentName",
  components: {
    "g-button": Button,
  },
}

# 使用

代码：

<template>
  <div>
    <g-button>默认按钮</g-button>
  </div>
</template>
